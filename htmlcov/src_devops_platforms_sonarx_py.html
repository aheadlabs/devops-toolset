19def get_quality_gate_status(properties_file_path: str, token: str, branch: str = None, pull_request: bool = False): 

20 """Gets the status of the quality gate of a project 

21 

22 Args: 

23 properties_file_path: Path to sonar-project.properties file. Keys 

24 sonar.projectKey and sonar.organization are mandatory. 

25 token: SonarCloud token used for authentication 

26 branch: Git branch name to get the quality gate for. If None, master 

27 will be assumed. 

28 pull_request: True if the analysis was originated by a pull request. 

29 """ 

30 

31 core.log_tools.log_indented_list(literals.get("function_params"), 

32 core.log_tools.get_parameter_value_list(locals()), 

33 core.log_tools.LogLevel.debug) 

34 

35 token_base64 = encode(f"{token}:") 

36 basic_auth_token = f"Basic {token_base64}" 

37 headers = {"Authorization": basic_auth_token} 

38 

39 logging.info(literals.get("sonar_getting_qg").format(branch=branch)) 

40 logging.info(literals.get("sonar_pr_mode").format(pull_request=pull_request)) 

41 branch_segment = generate_branch_segment(branch, pull_request) 

42 

43 logging.info(literals.get("sonar_config_file").format(file=properties_file_path)) 

44 sonar_url, sonar_project_key, sonar_organization = read_sonar_properties_file(properties_file_path) 

45 

46 url = f"{sonar_url}{Urls.SONAR_QUALITY_GATE_PARTIAL_URL}{sonar_project_key}{branch_segment}" 

47 logging.info(literals.get("sonar_qg_url").format(url=url)) 

48 

49 response = requests.get(url, headers=headers) 

50 if response.status_code != 200: 

51 raise ValueError(literals.get("sonar_unexpected_status_code").format(statusCode=response.status_code)) 

52 

53 quality_gate_data = response.json() 

54 logging.info(literals.get("sonar_qg_json").format(json=quality_gate_data)) 

55 

56 if quality_gate_data["projectStatus"]["status"] == "OK": 

57 logging.info(literals.get("sonar_qg_ok")) 

58 else: 

59 for condition in quality_gate_data["projectStatus"]["conditions"]: 

60 if condition["status"] == "ERROR": 

61 logging.error(literals.get("sonar_invalid_metric").format( 

62 metricKey=condition["metricKey"], 

63 actualValue=condition["actualValue"], 

64 comparator=condition["comparator"],

70def read_sonar_properties_file(path: str): 

71 """Reads a sonar-project.properties file 

72 

73 It adds a temporary dummy section to be correctly parsed by configParser 

74 

75 Args: 

76 path: Path to the sonar-project.properties file 

77 

78 Returns: 

79 Tuple containing url, projectKey, organization 

80 """ 

81 

82 with open(path, "r") as file: 

83 data = f"[temp]\n{file.read()}" 

84 

85 properties_file_parser = configparser.ConfigParser() 

86 properties_file_parser.read_string(data) 

87 

88 sonar_url = properties_file_parser.get("temp", "sonar.host.url") 

89 sonar_project_key = properties_file_parser.get("temp", "sonar.projectKey") 

90 sonar_organization = properties_file_parser.get("temp", "sonar.organization") 

91 

92 return sonar_url, sonar_project_key, sonar_organization
