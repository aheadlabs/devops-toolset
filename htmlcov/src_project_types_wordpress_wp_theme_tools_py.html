170 """Downloads a WordPress theme from a feed or a URL.

176 Args:

304def replace_theme_meta_data_in_package_file(file_path: str, src_theme_configuration: dict): 

305 """ From the theme configuration, creates a replacements dict and replaces the file_path 

306 Args: 

307 file_path: The path of the file to be replaced. 

308 src_theme_configuration: Dict containing src theme configuration.

310 replacements = {"name": src_theme_configuration["source"], "author": {}}

322 json_data = json.load(package_json_file)

354def replace_theme_meta_data(path: str, replacements: dict, regex_str: str): 

355 """ Opens the file path and rewrites it replacing matches from the replacements dict 

356 Args: 

357 path: The file path to be replaced. 

358 replacements: The replacements content. 

359 regex_str: The regex string used to match and replace the replacements dict. 

360 """ 

361 

362 replace_file = open(path, 'r', newline='\n') 

363 file_content = replace_file.read() 

364 

365 for key, replacement in replacements.items(): 

366 # Build the regex to retrieve data: This regex would match the lines containing [key]:value. 

367 # For example, in the string Author: Ahead Labs, S.L -> $1 = Author and $2 = Ahead Labs, S.L 

368 regex = f'({key}){regex_str}' 

369 

370 # Search with regex 

371 matches = re.search(regex, file_content) 

372 

373 if matches is not None and matches.group(1): 

374 # Build the replacement string 

375 target = f'{matches.group(1)}: {replacement}' 

376 # Apply the replacement 

377 file_content = file_content.replace(matches.group(0), target) 

378 

379 # Write replaced content to the file 

380 with open(path, 'w', newline='\n') as scss_file: 

381 scss_file.write(file_content)

467def triage_themes(themes: dict) -> (dict, dict): 

468 """triages themes to determine which must be installed and activated. 

469 

470 Args: 

471 themes: Themes configuration. 

472 

473 Return: 

474 Tuple with parent and child theme configuration. 

475 """ 

476 child = None 

477 parent = None 

478 parent_guess: str = "" 

479 

480 for theme in themes: 

481 if not theme["activate"]: 

482 parent_guess = theme["name"] 

483 if theme["activate"]: 

484 child = theme 

485 if theme["template"]: 

486 parent_guess = theme["template"] 

487 

488 if not theme["activate"] and not theme["template"] and theme["name"] == parent_guess: 

489 parent = theme 

490 

491 return parent, child
