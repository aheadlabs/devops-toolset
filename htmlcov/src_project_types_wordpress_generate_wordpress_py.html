188def generate_additional_wpconfig_files(site_config: dict, environments: dict, additional_environments: list, 

189 environments_db_user_passwords: dict, 

190 wordpress_path: str): 

191 """Generates additional wp-config.php files for different environments. 

192 

193 Args: 

194 site_config: Parsed site configuration file 

195 environments: All the available environment configurations. 

196 additional_environments: Additional environments to generate 

197 wp-config.php files for. 

198 environments_db_user_passwords: Additional environment db user 

199 passwords. 

200 wordpress_path: Path to the WordPress installation. 

201 """ 

202 

203 wordpress_path_obj = pathlib.Path(wordpress_path) 

204 wp_config_path = pathlib.Path.joinpath(wordpress_path_obj, "wp-config.php") 

205 wp_config_path_temp = pathlib.Path.joinpath(wordpress_path_obj, "wp-config-temp.php") 

206 

207 # Rename original file 

208 if paths.is_valid_path(wp_config_path.as_posix(), True): 

209 shutil.move(wp_config_path, wp_config_path_temp) 

210 

211 # Filter environments 

212 filtered_environments = list(filter(lambda environment_x: environment_x["name"] in additional_environments, 

213 environments)) 

214 

215 # Create additional configuration files for the filtered environments 

216 for environment in filtered_environments: 

217 project_types.wordpress.wptools.set_wordpress_config_from_configuration_file(site_config, environment, wordpress_path, 

218 environments_db_user_passwords[ 

219 environment["name"]]) 

220 shutil.move(wp_config_path, pathlib.Path.joinpath(wordpress_path_obj, f"wp-config-{environment['name']}.php")) 

221 

222 # Rename original file 

223 if paths.is_valid_path(str(wp_config_path_temp), True): 

224 shutil.move(wp_config_path_temp, wp_config_path)
